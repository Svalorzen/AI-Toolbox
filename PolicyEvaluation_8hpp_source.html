167  Values val0;
168  QFunction q = makeQFunction(S, A);
169  const auto p = policy.getPolicy();
170 
171  const bool useEpsilon = checkDifferentSmall(epsilon_, 0.0);
172  while ( timestep < horizon_ && (!useEpsilon || variation > epsilon_) ) {
173  ++timestep;

177
