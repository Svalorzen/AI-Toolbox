241  template <typename PIterator, typename VIterator, typename P = identity>
242  PIterator extractBestUsefulPoints(PIterator pbegin, PIterator pend, VIterator begin, VIterator end, P p = P{}) {
243  const auto pointsN = std::distance(pbegin, pend);
244  const auto entriesN = std::distance(begin, end);
245 
246  std::vector<std::pair<PIterator, double>> bestValues(entriesN, {pend, std::numeric_limits<double>::lowest()});
247  const auto maxBound = pointsN < entriesN ? pend : pbegin + entriesN;
248 
249  // So the idea here is that we advance IT only if we found a Point
250  // which supports a previously unsupported Hyperplane. This allows us
251  // to avoid doing later work for compacting the points before the
252  // bound.
253  //
254  // If instead the found Point takes into consideration an already
255  // supported Hyperplane, then it either is better or not. If it's
256  // better, we swap it with whatever was before. In both cases, the
257  // Point to discard ends up at the end and we decrease the bound.
258  auto it = pbegin;
259  auto bound = pend;
260  while (it < bound && it < maxBound) {
261  double value;
262  const auto vId = std::distance(begin, findBestAtPoint(*it, begin, end, &value, p));
263  if (bestValues[vId].second < value) {
264  if (bestValues[vId].first == pend) {
265  bestValues[vId] = {it++, value};
266  continue;
267  } else {
268  bestValues[vId].second = value;
269  std::iter_swap(bestValues[vId].first, it);
270  }
271  }
272  std::iter_swap(it, --bound);
273  }
274  if (it == bound) return it;
275 
276  // If all Hyperplanes have been supported by at least one Point, then
277  // we can finish up the rest with less swaps and checks. Here we only
278  // swap with the best if needed, otherwise we don't have to do
279  // anything.
280  //
281  // This is because we can return one Point per Hyperplane at the most,
282  // so if we're here the bound is not going to move anyway.
283  while (it < bound) {
284  double value;
285  const auto vId = std::distance(begin, findBestAtPoint(*it, begin, end, &value, p));
286  if (bestValues[vId].second < value) {
287  bestValues[vId].second = value;
288  std::iter_swap(bestValues[vId].first, it);
289  }
290  ++it;
291  }
292  return maxBound;
293  }

328  PointSurface findVerticesNaive(NewIt beginNew, NewIt endNew, OldIt alphasBegin, OldIt alphasEnd, P1 p1 = P1{}, P2 p2 = P2{}) {
329  PointSurface vertices;
330 
331  const size_t alphasSize = std::distance(alphasBegin, alphasEnd);
332  if (alphasSize == 0) return vertices;
333  const size_t S = std::invoke(p2, *alphasBegin).size();
334 
335  // This enumerator allows us to compute all possible subsets of S-1
336  // elements. We use it on both the alphas, and the boundaries, thus the
337  // number of elements we iterate over is alphasSize + S.
338  SubsetEnumerator enumerator(S - 1, 0ul, alphasSize + S);
339 
340  // This is the matrix on the left side of Ax = b (where A is m)
341  Matrix2D m(S + 1, S + 1);
342  m.row(0)[S] = -1; // First row is always a vector
343 
344  Vector boundary(S+1);
345  boundary[S] = 0.0; // The boundary doesn't care about the value
346 
347  // This is the vector on the right side of Ax = b
348  Vector b(S+1); b.setZero();
349 
350  Vector result(S+1);
351 
352  // Common matrix/vector setups
353 
354  for (auto newVIt = beginNew; newVIt != endNew; ++newVIt) {
355  m.row(0).head(S) = std::invoke(p1, *newVIt);
356 
357  enumerator.reset();
358 
359  // Get subset of planes, find corner with LU
360  size_t last = 0;
361  while (enumerator.isValid()) {
362  // Reset boundaries to care about all dimensions
363  boundary.head(S).fill(1.0);
364  size_t counter = 1;
365  // Note that we start from last to avoid re-copying vectors
366  // that are already in the matrix in their correct place.
367  for (auto i = last; i < enumerator->size(); ++i) {
368  // For each value in the enumerator, if it is less than
369  // alphasSize it is referring to an alphaVector we need to
370  // take into account.
371  const auto index = (*enumerator)[i];
372  if (index < alphasSize) {
373  // Copy the right vector in the matrix.
374  m.row(counter).head(S) = std::invoke(p2, *std::next(alphasBegin, index));
375  m.row(counter)[S] = -1;
376  ++counter;
377  } else {
378  // We limit the index-th dimension (minus alphasSize to scale in a 0-S range)
379  boundary[index - alphasSize] = 0.0;
380  }
381  }
382  m.row(counter) = boundary;
383  b[counter] = 1.0;
384  ++counter;
385 
386  // Note that we only need to consider the first "counter" rows,
387  // as the boundaries get merged in a single one.
388  result = m.topRows(counter).colPivHouseholderQr().solve(b.head(counter));
389 
390  b[counter-1] = 0.0;
391 
392  // Add to found only if valid, otherwise skip.
393  if (((result.head(S).array() >= 0) && (result.head(S).array() <= 1.0)).all()) {
394  vertices.first.emplace_back(result.head(S));
395  vertices.second.emplace_back(result[S]);
396  }
397 
398  // Advance, and take the id of the first index changed in the
399  // next combination.
400  last = enumerator.advance();
401 
402  // If the index went over the alpha list, then we'd only have
403  // boundaries, but we don't care about those cases (since we
404  // assume we already have the corners of the simplex computed).
405  // Thus, terminate.
406  if ((*enumerator)[last] >= alphasSize) break;
407  }
408  }
409  return vertices;
410  }
