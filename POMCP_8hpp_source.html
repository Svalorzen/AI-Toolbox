244  size_t POMCP<M>::sampleAction(size_t a, size_t o, unsigned horizon) {
245  auto & obs = graph_.children[a].children;
246 
247  auto it = obs.find(o);
248  if ( it == obs.end() )
249  return sampleAction(Belief(S, 1.0 / S), horizon);
250 
251  // Here we need an additional step, because *it is contained by graph_.
252  // If we just move assign, graph_ is first going to delete everything it
253  // contains (included *it), and then we are going to move unallocated memory
254  // into graph_! So we move *it outside of the graph_ hierarchy, so that
255  // we can then assign safely.
256  { auto tmp = std::move(it->second); graph_ = std::move(tmp); }
257

259  std::cerr << "POMCP Lost track of the belief, restarting with uniform..\n";

281  double POMCP<M>::simulate(BeliefNode & b, size_t s, unsigned depth) {
282  // Head update
283  if ( depth > 0 ) b.belief.push_back(s);
284  b.N++;
285 
286  auto begin = std::begin(b.children);
287  size_t a = std::distance(begin, findBestBonusA(begin, std::end(b.children), b.N));
288 
289  size_t s1, o; double rew;
290  std::tie(s1, o, rew) = model_.sampleSOR(s, a);
291 
292  auto & aNode = b.children[a];
293 
294  // We only go deeper if needed (maxDepth_ is always at least 1).
295  if ( depth + 1 < maxDepth_ && !model_.isTerminal(s1) ) {
296  auto end = std::end(aNode.children);
297  auto ot = aNode.children.find(o);
298 
299  double futureRew;
300  if ( ot == end ) {
301  aNode.children.emplace(std::piecewise_construct,
302  std::forward_as_tuple(o),
303  std::forward_as_tuple(s1, A));
304  futureRew = rollout(s1, depth + 1);
305  }
306  else {
307  futureRew = simulate( ot->second, s1, depth + 1 );
308  }
309 
310  rew += model_.getDiscount() * futureRew;
311  }
312 
313  // Action update
314  aNode.N++;
315  aNode.V += ( rew - aNode.V ) / static_cast<double>(aNode.N);
316 
317  return rew;
318  }

341  template <typename Iterator>
342  Iterator POMCP<M>::findBestBonusA(Iterator begin, Iterator end, unsigned count) {
343  // Count here can be as low as 1.
344  // Since log(1) = 0, and 0/0 = error, we add 1.0.
345  double logCount = std::log(count + 1.0);
346  // We use this function to produce a score for each action. This can be easily
347  // substituted with something else to produce different POMCP variants.
348  auto evaluationFunction = [this, logCount](const ActionNode & an){
349  return an.V + exploration_ * std::sqrt( logCount / an.N );
350  };
351 
352  auto bestIterator = begin++;
353  double bestValue = evaluationFunction(*bestIterator);
354 
355  for ( ; begin < end; ++begin ) {
356  double actionValue = evaluationFunction(*begin);
357  if ( actionValue > bestValue ) {
358  bestValue = actionValue;
359  bestIterator = begin;
360  }
361  }
362 
363  return bestIterator;
364  }
