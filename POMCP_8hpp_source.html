332  size_t POMCP<M>::sampleAction(const size_t a, const size_t o, const unsigned horizon) {
333  const auto & obs = graph_.children[a].children;
334 
335  auto it = obs.find(o);
336  if ( it == obs.end() ) {
337  AI_LOGGER(AI_SEVERITY_WARNING, "Observation " << o << " never experienced in simulation, restarting with uniform belief..");
338  auto b = Belief(S); b.fill(1.0/S);
339  return sampleAction(b, horizon);
340  }
341 
342  // Here we need an additional step, because *it is contained by graph_.
343  // If we just move assign, graph_ is first going to delete everything it
344  // contains (included *it), and then we are going to move unallocated memory
345  // into graph_! So we move *it outside of the graph_ hierarchy, so that
346  // we can then assign safely.
347  { auto tmp = std::move(it->second); graph_ = std::move(tmp); }
348 
349  if ( ! graph_.belief.size() ) {
350  AI_LOGGER(AI_SEVERITY_WARNING, "POMCP lost track of the belief, restarting with uniform..");
351  auto b = Belief(S); b.fill(1.0/S);
352  return sampleAction(b, horizon);
353  }
354 
355  // We resize here in case we didn't have time to sample the new
356  // head node. In this case, the new head may not have children.
357  // This would break the UCT call.
358  graph_.children.resize(A);
359 
360  return runSimulation(horizon);

377  template <typename M>
378  double POMCP<M>::simulate(BeliefNode & b, const size_t s, const unsigned depth) {
379  b.N++;
380 
381  auto begin = std::begin(b.children);
382  const size_t a = std::distance(begin, findBestBonusA(begin, std::end(b.children), b.N));
383 
384  auto [s1, o, rew] = model_.sampleSOR(s, a);
385 
386  auto & aNode = b.children[a];
387 
388  {
389  double futureRew = 0.0;
390  // We need to append the node anyway to perform the belief
391  // update for the next timestep.
392  auto ot = aNode.children.find(o);
393  if ( ot == std::end(aNode.children) ) {
394  aNode.children.emplace(std::piecewise_construct,
395  std::forward_as_tuple(o),
396  std::forward_as_tuple(s1));
397  // This stops automatically if we go out of depth
398  futureRew = rollout(s1, depth + 1);
399  }
400  else {
401  ot->second.belief.push_back(s1);
402  // We only go deeper if needed (maxDepth_ is always at least 1).
403  if ( depth + 1 < maxDepth_ && !model_.isTerminal(s1) ) {
404  // Since most memory is allocated on the leaves,
405  // we do not allocate on node creation but only when
406  // we are actually descending into a node. If the node
407  // already has memory this should not do anything in
408  // any case.
409  ot->second.children.resize(A);
410  futureRew = simulate( ot->second, s1, depth + 1 );
411  }
412  }
413 
414  rew += model_.getDiscount() * futureRew;
415  }
416 
417  // Action update
418  aNode.N++;
419  aNode.V += ( rew - aNode.V ) / static_cast<double>(aNode.N);
420 
421  return rew;
422  }

425  double POMCP<M>::rollout(size_t s, unsigned depth) {
426  double rew = 0.0, totalRew = 0.0, gamma = 1.0;
427 
428  std::uniform_int_distribution<size_t> generator(0, A-1);
429  for ( ; depth < maxDepth_; ++depth ) {
430  std::tie( s, rew ) = model_.sampleSR( s, generator(rand_) );
431  totalRew += gamma * rew;
432 
433  if (model_.isTerminal(s))
434  return totalRew;
435 
436  gamma *= model_.getDiscount();
437  }
438  return totalRew;

448  template <typename Iterator>
449  Iterator POMCP<M>::findBestBonusA(Iterator begin, Iterator end, const unsigned count) {
450  // Count here can be as low as 1.
451  // Since log(1) = 0, and 0/0 = error, we add 1.0.
452  const double logCount = std::log(count + 1.0);
453  // We use this function to produce a score for each action. This can be easily
454  // substituted with something else to produce different POMCP variants.
455  auto evaluationFunction = [this, logCount](const ActionNode & an){
456  return an.V + exploration_ * std::sqrt( logCount / an.N );
457  };
458 
459  auto bestIterator = begin++;
460  double bestValue = evaluationFunction(*bestIterator);
461 
462  for ( ; begin < end; ++begin ) {
463  const double actionValue = evaluationFunction(*begin);
464  if ( actionValue > bestValue ) {
465  bestValue = actionValue;
466  bestIterator = begin;
467  }
468  }
469 
470  return bestIterator;
471  }
