331  size_t POMCP<M>::sampleAction(const size_t a, const size_t o, const unsigned horizon) {
332  const auto & obs = graph_.children[a].children;
333 
334  auto it = obs.find(o);
335  if ( it == obs.end() ) {
336  AI_LOGGER(AI_SEVERITY_WARNING, "Observation " << o << " never experienced in simulation, restarting with uniform belief..");
337  auto b = Belief(S); b.fill(1.0/S);
338  return sampleAction(b, horizon);
339  }
340 
341  // Here we need an additional step, because *it is contained by graph_.
342  // If we just move assign, graph_ is first going to delete everything it
343  // contains (included *it), and then we are going to move unallocated memory
344  // into graph_! So we move *it outside of the graph_ hierarchy, so that
345  // we can then assign safely.
346  { auto tmp = std::move(it->second); graph_ = std::move(tmp); }
347 
348  if ( ! graph_.belief.size() ) {
349  AI_LOGGER(AI_SEVERITY_WARNING, "POMCP Lost track of the belief, restarting with uniform..");
350  auto b = Belief(S); b.fill(1.0/S);
351  return sampleAction(b, horizon);
352  }
353 
354  // We resize here in case we didn't have time to sample the new
355  // head node. In this case, the new head may not have children.
356  // This would break the UCT call.
357  graph_.children.resize(A);

376  template <typename M>
377  double POMCP<M>::simulate(BeliefNode & b, const size_t s, const unsigned depth) {
378  b.N++;
379 
380  auto begin = std::begin(b.children);
381  const size_t a = std::distance(begin, findBestBonusA(begin, std::end(b.children), b.N));
382 
383  auto [s1, o, rew] = model_.sampleSOR(s, a);
384 
385  auto & aNode = b.children[a];
386 
387  {
388  double futureRew = 0.0;
389  // We need to append the node anyway to perform the belief
390  // update for the next timestep.
391  auto ot = aNode.children.find(o);
392  if ( ot == std::end(aNode.children) ) {
393  aNode.children.emplace(std::piecewise_construct,
394  std::forward_as_tuple(o),
395  std::forward_as_tuple(s1));
396  // This stops automatically if we go out of depth
397  futureRew = rollout(s1, depth + 1);
398  }
399  else {
400  ot->second.belief.push_back(s1);
401  // We only go deeper if needed (maxDepth_ is always at least 1).
402  if ( depth + 1 < maxDepth_ && !model_.isTerminal(s1) ) {
403  // Since most memory is allocated on the leaves,
404  // we do not allocate on node creation but only when
405  // we are actually descending into a node. If the node
406  // already has memory this should not do anything in
407  // any case.
408  ot->second.children.resize(A);
409  futureRew = simulate( ot->second, s1, depth + 1 );
410  }
411  }
412 
413  rew += model_.getDiscount() * futureRew;
414  }
415 
416  // Action update
417  aNode.N++;
418  aNode.V += ( rew - aNode.V ) / static_cast<double>(aNode.N);
419 
420  return rew;
421  }

444  template <typename Iterator>
445  Iterator POMCP<M>::findBestBonusA(Iterator begin, Iterator end, const unsigned count) {
446  // Count here can be as low as 1.
447  // Since log(1) = 0, and 0/0 = error, we add 1.0.
448  const double logCount = std::log(count + 1.0);
449  // We use this function to produce a score for each action. This can be easily
450  // substituted with something else to produce different POMCP variants.
451  auto evaluationFunction = [this, logCount](const ActionNode & an){
452  return an.V + exploration_ * std::sqrt( logCount / an.N );
453  };
454 
455  auto bestIterator = begin++;
456  double bestValue = evaluationFunction(*bestIterator);
457 
458  for ( ; begin < end; ++begin ) {
459  const double actionValue = evaluationFunction(*begin);
460  if ( actionValue > bestValue ) {
461  bestValue = actionValue;
462  bestIterator = begin;
463  }
464  }
465 
466  return bestIterator;
467  }
