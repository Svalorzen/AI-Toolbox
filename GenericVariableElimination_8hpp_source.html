190  template <typename Global>
191  void GenericVariableElimination<Factor>::removeFactor(const Factors & F, Graph & graph, const size_t f, FinalFactors & finalFactors, Global & global) {
192  AI_LOGGER(AI_SEVERITY_DEBUG, "Removing factor " << f);
193 
194  // We iterate over all possible joint values of the neighbors of 'f';
195  // these are all variables which share at least one factor with it.
196  const auto & factors = graph.getFactors(f);
197  const auto & vNeighbors = graph.getVariables(f);
198 
199  if constexpr(global_interface<Global>::beginRemoval)
200  Impl::callFunction(global, &Global::beginRemoval, graph, factors, vNeighbors, f);
201 
202  // We'll now create new rules that represent the elimination of the
203  // input variable for this round.
204  const bool isFinalFactor = vNeighbors.size() == 0;
205 
206  Rules * oldRulesP;
207  size_t oldRulesCurrId = 0;
208 
209  PartialFactorsEnumerator jointValues(F, vNeighbors, f, true);
210  const auto id = jointValues.getFactorToSkipId();
211 
212  if (!isFinalFactor) {
213  oldRulesP = &graph.getFactor(vNeighbors)->getData();
214  oldRulesP->reserve(jointValues.size());
215  }
216 
217  AI_LOGGER(
218  AI_SEVERITY_INFO,
219  "Width of this factor: " << vNeighbors.size() + 1 << ". "
220  "Joint values to iterate: " << jointValues.size() * F[f]
221  );
222 
223  size_t jvID = 0;
224  while (jointValues.isValid()) {
225  auto & jointValue = *jointValues;
226 
227  if constexpr(global_interface<Global>::initNewFactor)
228  global.initNewFactor();
229 
230  // Since we are eliminating 'f', we iterate over its possible
231  // values and we reduce over them; this could be a cross-sum
232  // operation, a max, or anything else.
233  for (size_t fValue = 0; fValue < F[f]; ++fValue) {
234  if constexpr(global_interface<Global>::beginCrossSum)
235  Impl::callFunction(global, &Global::beginCrossSum, fValue);
236 
237  jointValue.second[id] = fValue;
238  for (const auto factor : factors) {
239  if constexpr(global_interface<Global>::beginFactorCrossSum)
240  global.beginFactorCrossSum();
241 
242  // We reduce over each Factor that is applicable to this
243  // particular joint value set.
244  const size_t jvPartialIndex = toIndexPartial(factor->getVariables(), F, jointValue);
245  if constexpr(global_interface<Global>::mergeFactors) {
246  const auto & data = factor->getData();
247  const auto ruleIt = std::lower_bound(
248  std::begin(data),
249  std::end(data),
250  jvPartialIndex,
251  [](const Rule & lhs, const size_t rhs) {
252  return lhs.first < rhs;
253  }
254  );
255  if (ruleIt != std::end(data) && ruleIt->first == jvPartialIndex)
256  global.crossSum(ruleIt->second);
257  } else {
258  for (const auto rule : factor->getData())
259  if (jvPartialIndex == rule.first)
260  global.crossSum(rule.second);
261  }
262 
263  if constexpr(global_interface<Global>::endFactorCrossSum)
264  global.endFactorCrossSum();
265  }
266 
267  if constexpr(global_interface<Global>::endCrossSum)
268  global.endCrossSum();
269  }
270 
271  bool isValidNewFactor = true;
272  if constexpr(global_interface<Global>::isValidNewFactor)
273  isValidNewFactor = global.isValidNewFactor();
274 
275  // If the new Factor is good, we save it together with the joint
276  // value that has produced it (minus the one of the variable to
277  // remove). If it has no neighbors, we add it to the finalFactors
278  // instead.
279  if (isValidNewFactor) {
280  if (!isFinalFactor) {
281  auto & oldRules = *oldRulesP;
282 
283  // If we care enough to merge, we store all rules in
284  // lexicographical order of value; if the old rules already
285  // contained this same value and we are provided with a
286  // merge function, we can merge the two, otherwise we
287  // insert it as-is in the correct spot.
288  if constexpr(global_interface<Global>::mergeFactors) {
289  while (oldRulesCurrId < oldRules.size() && oldRules[oldRulesCurrId].first < jvID)
290  ++oldRulesCurrId;
291 
292  if (oldRulesCurrId < oldRules.size() && oldRules[oldRulesCurrId].first == jvID) {
293  global.mergeFactors(oldRules[oldRulesCurrId].second, std::move(global.newFactor));
294  } else {
295  oldRules.emplace(std::begin(oldRules) + oldRulesCurrId, jvID, std::move(global.newFactor));
296  }
297  } else {
298  // Otherwise we simply append, as it should be faster.
299  // Remember, a factor may be appended on multiple
300  // times, but it's only iterated over once before being
301  // removed.
302  oldRules.emplace_back(jvID, std::move(global.newFactor));
303  }
304  ++oldRulesCurrId;
305  }
306  else
307  finalFactors.push_back(std::move(global.newFactor));
308  }
309  ++jvID;
310  jointValues.advance();
311  }
312 
313  // And finally as usual in variable elimination remove the variable
314  // from the graph and insert the newly created variable in.
315  graph.erase(f);
316  }
