8  // The following problem is the AAAI-94 Tiger problem, with
9  // a 0.95 discount factor. The problem can be stated as follows:
10  //
11  // The agent stands in front of two doors. He can open either of
12  // them. Behind one door, there is an agent-eater tiger, and
13  // in the other a small treasure. The agent does not know to what
14  // each door leads to, but instead of just opening the door, he
15  // can listen. When he listens, it will hear the tiger from either
16  // the left or right door. Its hearing is imperfect though, and
17  // 15% of the time it will hear the tiger behind the wrong door.
18  //
19  // Once the agent opens a door, it will either get a great penalty
20  // due to being eaten by the tiger, or get the reward. After that
21  // the game will automatically reset to an unknown state: the agent
22  // needs to start guessing again where the new tiger and treasure
23  // are.
24  //
25  // The states here are the positions of the tiger and treasure:
26  // since there are two doors, there are two states.
27  //
28  // There are three actions, corresponding to the listen action and
29  // open door actions.
30  //
31  // There are two possible observations, which are always random but
32  // for the listen action: in that case, we will obtain the correct
33  // information about the true state 85% of the time.
34  //
35  // The solutions of this problem have been computed using Tony
36  // Cassandra's pomdp-solve program (www.pomdp.org).
37

53  inline AIToolbox::POMDP::Model<AIToolbox::MDP::Model> makeTigerProblem() {
54  using namespace TigerProblemEnums;
55 
56  // Actions are: 0-listen, 1-open-left, 2-open-right
57  size_t S = 2, A = 3, O = 2;
58 
59  AIToolbox::POMDP::Model<AIToolbox::MDP::Model> model(O, S, A);
60 
61  AIToolbox::DumbMatrix3D transitions(boost::extents[S][A][S]);
62  AIToolbox::DumbMatrix3D rewards(boost::extents[S][A][S]);
63  AIToolbox::DumbMatrix3D observations(boost::extents[S][A][O]);
64 
65  // Transitions
66  // If we listen, nothing changes.
67  for ( size_t s = 0; s < S; ++s )
68  transitions[s][A_LISTEN][s] = 1.0;
69 
70  // If we pick a door, tiger and treasure shuffle.
71  for ( size_t s = 0; s < S; ++s ) {
72  for ( size_t s1 = 0; s1 < S; ++s1 ) {
73  transitions[s][A_LEFT ][s1] = 1.0 / S;
74  transitions[s][A_RIGHT][s1] = 1.0 / S;
75  }
76  }
77 
78  // Observations
79  // If we listen, we guess right 85% of the time.
80  observations[TIG_LEFT ][A_LISTEN][TIG_LEFT ] = 1.0 - listenError;
81  observations[TIG_LEFT ][A_LISTEN][TIG_RIGHT] = listenError;
82 
83  observations[TIG_RIGHT][A_LISTEN][TIG_RIGHT] = 1.0 - listenError;
84  observations[TIG_RIGHT][A_LISTEN][TIG_LEFT ] = listenError;
85 
86  // Otherwise we get no information on the environment.
87  for ( size_t s = 0; s < S; ++s ) {
88  for ( size_t o = 0; o < O; ++o ) {
89  observations[s][A_LEFT ][o] = 1.0 / O;
90  observations[s][A_RIGHT][o] = 1.0 / O;
91  }
92  }
93 
94  // Rewards
95  // Listening has a small penalty
96  for ( size_t s = 0; s < S; ++s )
97  for ( size_t s1 = 0; s1 < S; ++s1 )
98  rewards[s][A_LISTEN][s1] = -1.0;
99 
100  // Treasure has a decent reward, and tiger a bad penalty.
101  for ( size_t s1 = 0; s1 < S; ++s1 ) {
102  rewards[TIG_RIGHT][A_LEFT][s1] = 10.0;
103  rewards[TIG_LEFT ][A_LEFT][s1] = -100.0;
104 
105  rewards[TIG_LEFT ][A_RIGHT][s1] = 10.0;
106  rewards[TIG_RIGHT][A_RIGHT][s1] = -100.0;
107  }
108 
109  model.setTransitionFunction(transitions);
110  model.setRewardFunction(rewards);
111  model.setObservationFunction(observations);
112 
113  return model;
114  }
115 }
