104  CooperativePrioritizedSweeping<M>::CooperativePrioritizedSweeping(const M & m, std::vector<std::vector<size_t>> basisDomains, double alpha, double theta) :
105  model_(m),
106  alpha_(alpha), theta_(theta),
107  qDomains_(std::move(basisDomains)),
108  rewardWeights_(model_.getS().size()),
109  deltaStorage_(model_.getS().size()),
110  gp_(model_.getS(), model_.getA(), q_),
111  queue_(model_.getGraph()),
112  rand_(Impl::Seeder::getSeed())
113  {
114  // We weight the rewards so that they are split correctly between the
115  // components of the QFunction.
116  // Note that unused reward weights might result in r/0 or 0/0
117  // operations, but since then we won't be using those elements
118  // anyway it's not a problem.
119  rewardWeights_.setZero();
120  deltaStorage_.setZero();
121 
122  const auto & nodes = model_.getGraph().getNodes();
123 
124  q_.bases.reserve(qDomains_.size());
125  for (const auto & domain : qDomains_) {
126  q_.bases.emplace_back();
127  auto & fm = q_.bases.back();
128 
129  for (auto d : domain) {
130  // Note that there's one more Q factor that depends
131  // this state factor.
132  rewardWeights_[d] += 1.0;
133 
134  // Compute state-action domain for this Q factor.
135  fm.actionTag = merge(fm.actionTag, nodes[d].agents);
136  for (const auto & n : nodes[d].parents)
137  fm.tag = merge(fm.tag, n);
138  }
139 
140  // Initialize this factor's matrix.
141  const size_t sizeA = factorSpacePartial(fm.actionTag, model_.getA());
142  const size_t sizeS = factorSpacePartial(fm.tag, model_.getS());
143 
144  fm.values.resize(sizeS, sizeA);
145  fm.values.setZero();
146  }

156  void CooperativePrioritizedSweeping<M>::batchUpdateQ(const unsigned N) {
157  // Initialize some variables to avoid reallocations
158  State s(model_.getS().size());
159  State s1(model_.getS().size());
160  Action a(model_.getA().size());
161  Rewards rews(model_.getS().size());
162 
163  for (size_t n = 0; n < N; ++n) {
164  if (!queue_.getNonZeroPriorities()) return;
165 
166  queue_.reconstruct(s, a);
167 
168  // Filling randomly the missing elements.
169  for (size_t i = 0; i < s.size(); ++i) {
170  if (s[i] == model_.getS()[i]) {
171  std::uniform_int_distribution<size_t> dist(0, model_.getS()[i]-1);
172  s[i] = dist(rand_);
173  }
174  }
175  for (size_t i = 0; i < a.size(); ++i) {
176  if (a[i] == model_.getA()[i]) {
177  std::uniform_int_distribution<size_t> dist(0, model_.getA()[i]-1);
178  a[i] = dist(rand_);
179  }
180  }
181 
182  // Finally, sample a new s1/rews from the model.
183  model_.sampleSRs(s, a, &s1, &rews);
184  rews.array() /= rewardWeights_.array();
185 
186  // And use them to update Q.
187  updateQ(s, a, s1, rews);
188 
189  // Update the queue
190  addToQueue(s);
191  }

200  void CooperativePrioritizedSweeping<M>::updateQ(const State & s, const Action & a, const State & s1, const Rewards & r) {
201  // Compute optimal action to do Q-Learning update.
202  const auto a1 = gp_.sampleAction(s1);
203 
204  // We update each Q factor separately.
205  for (size_t i = 0; i < q_.bases.size(); ++i) {
206  auto & q = q_.bases[i];
207 
208  const auto sid = toIndexPartial(q.tag, model_.getS(), s);
209  const auto aid = toIndexPartial(q.actionTag, model_.getA(), a);
210 
211  const auto s1id = toIndexPartial(q.tag, model_.getS(), s1);
212  const auto a1id = toIndexPartial(q.actionTag, model_.getA(), a1);
213 
214  // Compute numerical reward from the components children of this Q
215  // factor.
216  double rr = 0.0;
217  for (auto s : qDomains_[i])
218  rr += r[s]; // already divided by weights
219 
220  const auto originalQ = q.values(sid, aid);
221 
222  // Q-Learning update
223  q.values(sid, aid) += alpha_ * ( rr + model_.getDiscount() * q.values(s1id, a1id) - q.values(sid, aid) );
224 
225  // Split the delta to each element referenced by this Q factor.
226  // Note that we add to the storage, which is only cleared once we
227  // call addToQueue; this means that multiple calls to this
228  // functions cumulate their deltas.
229  const auto delta = std::fabs(originalQ - q.values(sid, aid)) / q.tag.size();
230  for (auto s : q.tag)
231  deltaStorage_[s] += delta;
232  }
233  }

235  template <typename M>
236  void CooperativePrioritizedSweeping<M>::addToQueue(const State & s1) {
237  // Note that s1 was s before, but here we consider it as the
238  // "future" state as we look for its parents.
239  const auto & T = model_.getTransitionFunction();
240  const auto & graph = model_.getGraph();
241 
242  for (size_t i = 0; i < s1.size(); ++i) {
243  // If the delta to apply is very small, we don't bother with it yet.
244  // This allows us to save some work until it's actually worth it.
245  if (deltaStorage_[i] < queue_.getNodeMaxPriority(i)) continue;
246 
247  // Here we need to iterate over j, but the queue still needs the
248  // a,s variables. So we keep all of them in mind to keep things easy.
249  size_t j = 0;
250  for (size_t a = 0; a < graph.getPartialSize(i); ++a) {
251  for (size_t s = 0; s < graph.getPartialSize(i, a); ++s) {
252  const auto p = T.transitions[i](j, s1[i]) * deltaStorage_[i];
253 
254  // Increase j before we check if we want to skip.
255  ++j;
256  // If it's not large enough, skip it.
257  if (p < theta_) continue;
258 
259  queue_.update(i, a, s, p);
260  }
261  }
262  // Reset this delta.
263  deltaStorage_[i] = 0.0;
264  }
265  }
