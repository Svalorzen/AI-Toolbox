110  CooperativePrioritizedSweeping<M>::CooperativePrioritizedSweeping(const M & m, std::vector<std::vector<size_t>> basisDomains, double alpha, double theta) :
111  model_(m),
112  alpha_(alpha), theta_(theta),
113  qDomains_(std::move(basisDomains)),
114  rewardWeights_(model_.getS().size()),
115  deltaStorage_(model_.getS().size()),
116  rewardStorage_(model_.getS().size()),
117  gp_(model_.getS(), model_.getA(), q_),
118  queue_(model_.getGraph()),
119  rand_(Impl::Seeder::getSeed())
120  {
121  // We weight the rewards so that they are split correctly between the
122  // components of the QFunction.
123  // Note that unused reward weights might result in r/0 or 0/0
124  // operations, but since then we won't be using those elements
125  // anyway it's not a problem.
126  rewardWeights_.setZero();
127  deltaStorage_.setZero();
128  // We don't need to zero rewardStorage_
129 
130  const auto & ps = model_.getGraph().getParentSets();
131 
132  q_.bases.reserve(qDomains_.size());
133  for (const auto & domain : qDomains_) {
134  q_.bases.emplace_back();
135  auto & q = q_.bases.back();
136 
137  for (const auto d : domain) {
138  // Compute state-action domain for this Q factor.
139  q.actionTag = merge(q.actionTag, ps[d].agents);
140  for (const auto & n : ps[d].features)
141  q.tag = merge(q.tag, n);
142  }
143  // We weight rewards based on the state features of each Q factor
144  for (const auto d : q.tag)
145  rewardWeights_[d] += 1.0;
146 
147  // Initialize this factor's matrix.
148  const size_t sizeA = factorSpacePartial(q.actionTag, model_.getA());
149  const size_t sizeS = factorSpacePartial(q.tag, model_.getS());
150 
151  q.values.resize(sizeS, sizeA);
152  q.values.setZero();
153  }

163  void CooperativePrioritizedSweeping<M>::batchUpdateQ(const unsigned N) {
164  // Initialize some variables to avoid reallocations
165  State s(model_.getS().size());
166  State s1(model_.getS().size());
167  Action a(model_.getA().size());
168  Rewards rews(model_.getS().size());
169 
170  for (size_t n = 0; n < N; ++n) {
171  if (!queue_.getNonZeroPriorities()) return;
172 
173  queue_.reconstruct(s, a);
174 
175  // Filling randomly the missing elements.
176  for (size_t i = 0; i < s.size(); ++i) {
177  if (s[i] == model_.getS()[i]) {
178  std::uniform_int_distribution<size_t> dist(0, model_.getS()[i]-1);
179  s[i] = dist(rand_);
180  }
181  }
182  for (size_t i = 0; i < a.size(); ++i) {
183  if (a[i] == model_.getA()[i]) {
184  std::uniform_int_distribution<size_t> dist(0, model_.getA()[i]-1);
185  a[i] = dist(rand_);
186  }
187  }
188 
189  // Finally, sample a new s1/rews from the model.
190  model_.sampleSRs(s, a, &s1, &rews);
191 
192  // And use them to update Q.
193  updateQ(s, a, s1, rews);
194 
195  // Update the queue
196  addToQueue(s);
197  }

200  template <typename M>
201  void CooperativePrioritizedSweeping<M>::updateQ(const State & s, const Action & a, const State & s1, const Rewards & r) {
202  // Compute optimal action to do Q-Learning update.
203  const auto a1 = gp_.sampleAction(s1);
204 
205  // The standard Q-update is in the form:
206  //
207  // Q(s,a) += alpha * ( R(s,a) + gamma * Q(s', a') - Q(s,a) )
208  //
209  // Since our Q-function is factored, we want to split the rewards per
210  // state feature (similar to SparseCooperativeQLearning).
211 
212  // Start with R
213  rewardStorage_ = r.array();
214  // Now go over the factored Q-function for the rest
215  for (const auto & q : q_.bases) {
216  const auto sid = toIndexPartial(q.tag, model_.getS(), s);
217  const auto aid = toIndexPartial(q.actionTag, model_.getA(), a);
218 
219  const auto s1id = toIndexPartial(q.tag, model_.getS(), s1);
220  const auto a1id = toIndexPartial(q.actionTag, model_.getA(), a1);
221 
222  // gamma * Q(s', a') - Q(s, a)
223  // We normalize it per state features, since we distribute the diff to all
224  // elements of rewardStorage_.
225  const auto diff = (model_.getDiscount() * q.values(s1id, a1id) - q.values(sid, aid)) / q.tag.size();
226 
227  // Apply the values to each state feature that applies to this Q factor.
228  // R(s,a) + ...
229  for (const auto s : q.tag)
230  rewardStorage_[s] += diff;
231  }
232 
233  // Normalize all values based on Q-factors
234  rewardStorage_.array() /= rewardWeights_.array();
235  rewardStorage_.array() *= alpha_;
236 
237  // We update each Q factor separately.
238  for (size_t i = 0; i < q_.bases.size(); ++i) {
239  auto & q = q_.bases[i];
240 
241  const auto sid = toIndexPartial(q.tag, model_.getS(), s);
242  const auto aid = toIndexPartial(q.actionTag, model_.getA(), a);
243 
244  // Compute numerical reward from the components children of this Q
245  // factor.
246  double td = 0.0;
247  for (const auto s : q.tag)
248  td += rewardStorage_[s];
249 
250  q.values(sid, aid) += td;
251 
252  // Split the delta to each element referenced by this Q factor.
253  // Note that we add to the storage, which is only cleared once we
254  // call addToQueue; this means that multiple calls to this
255  // functions cumulate their deltas.
256  const auto delta = std::fabs(td) / q.tag.size();
257  for (const auto s : q.tag)
258  deltaStorage_[s] += delta;
259  }
260  }

263  void CooperativePrioritizedSweeping<M>::addToQueue(const State & s1) {
264  // Note that s1 was s before, but here we consider it as the
265  // "future" state as we look for its parents.
266  const auto & T = model_.getTransitionFunction();
267  const auto & graph = model_.getGraph();
268 
269  for (size_t i = 0; i < s1.size(); ++i) {
270  // If the delta to apply is very small, we don't bother with it yet.
271  // This allows us to save some work until it's actually worth it.
272  if (deltaStorage_[i] < queue_.getNodeMaxPriority(i)) continue;
273 
274  // Here we need to iterate over j, but the queue still needs the
275  // a,s variables. So we keep all of them in mind to keep things easy.
276  size_t j = 0;
277  for (size_t a = 0; a < graph.getPartialSize(i); ++a) {
278  for (size_t s = 0; s < graph.getPartialSize(i, a); ++s) {
279  const auto p = T.transitions[i](j, s1[i]) * deltaStorage_[i];
280 
281  // Increase j before we check if we want to skip.
282  ++j;
283  // If it's not large enough, skip it.
284  if (p < theta_) continue;
285 
286  queue_.update(i, a, s, p);
287  }
288  }
289  // Reset this delta.
290  deltaStorage_[i] = 0.0;
291  }
292  }
