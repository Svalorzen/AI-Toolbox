558  std::tuple<size_t, double> bestConservativeAction(const M & pomdp, MDP::QFunction immediateRewards, const Belief & initialBelief, const VList & lbVList, MDP::Values * alpha = nullptr) {
559  // Note that we update inline the alphavectors in immediateRewards
560  Vector bpAlpha(pomdp.getS());
561  // Storage to avoid reallocations
562  Belief intermediateBelief(pomdp.getS());
563  Belief nextBelief(pomdp.getS());
564 
565  for (size_t a = 0; a < pomdp.getA(); ++a) {
566  updateBeliefPartial(pomdp, initialBelief, a, &intermediateBelief);
567 
568  bpAlpha.setZero();
569 
570  for (size_t o = 0; o < pomdp.getO(); ++o) {
571  updateBeliefPartialUnnormalized(pomdp, intermediateBelief, a, o, &nextBelief);
572 
573  const auto nextBeliefProbability = nextBelief.sum();
574  if (checkEqualSmall(nextBeliefProbability, 0.0)) continue;
575  // Now normalized
576  nextBelief /= nextBeliefProbability;
577 
578  const auto it = findBestAtPoint(nextBelief, std::begin(lbVList), std::end(lbVList), nullptr, unwrap);
579 
580  bpAlpha += pomdp.getObservationFunction(a).col(o).cwiseProduct(it->values);
581  }
582  immediateRewards.col(a) += pomdp.getDiscount() * pomdp.getTransitionFunction(a) * bpAlpha;
583  }
584 
585  size_t id;
586  double v = (initialBelief.transpose() * immediateRewards).maxCoeff(&id);
587 
588  // Copy alphavector for selected action if needed
589  if (alpha) *alpha = immediateRewards.col(id);
590 
591  return std::make_tuple(id, v);
592  }

637  else

639  }
640  qvals[a] += pomdp.getDiscount() * sum;
641  }
642  size_t bestAction;

644
