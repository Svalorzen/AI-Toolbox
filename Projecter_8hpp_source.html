91  typename Projecter<M>::ProjectionsRow Projecter<M>::operator()(const VList & w, const size_t a) {
92  ProjectionsRow projections( boost::extents[O] );
93 
94  for ( size_t o = 0; o < O; ++o ) {
95  // Here we put in just the immediate rewards so that the cross-summing step in the main
96  // function works correctly. However we communicate via the boolean that pruning should
97  // not be done at this step (since adding constants shouldn't do anything anyway).
98  if ( !possibleObservations_[a][o] ) {
99  // We add a parent id anyway in order to keep the code that cross-sums simple. However
100  // note that this fake ID of 0 should never be used, so it should be safe to avoid
101  // setting it to a special value like -1. If one really wants to check, he/she can
102  // just look at the observation table and the belief and see if it makes sense.
103  projections[o].emplace_back(immediateRewards_.row(a), a, VObs(1,0));
104  continue;
105  }
106 
107  // Otherwise we compute a projection for each ValueFunction supplied to us.
108  MDP::Values vproj(S);
109  for ( size_t i = 0; i < w.size(); ++i ) {
110  const auto & v = w[i].values;
111  // For each value function in the previous timestep, we compute the new value
112  // if we performed action a and obtained observation o.
113  // vproj_{a,o}[s] = R(s,a) / |O| + discount * sum_{s'} ( T(s,a,s') * O(s',a,o) * v_{t-1}(s') )
114  if constexpr(is_model_eigen_v<M>) {
115  vproj = model_.getTransitionFunction(a) * (v.cwiseProduct(model_.getObservationFunction(a).col(o)));
116  } else {
117  vproj.fill(0.0);
118  for ( size_t s = 0; s < S; ++s )
119  for ( size_t s1 = 0; s1 < S; ++s1 )
120  vproj[s] += model_.getTransitionProbability(s,a,s1) * model_.getObservationProbability(s1,a,o) * v[s1];
121  }
122  // Set new projection with found value and previous V id.
123  // projections[o].emplace_back(vproj, a, VObs(1,i));
124  projections[o].emplace_back(vproj * discount_ + immediateRewards_.row(a).transpose(), a, VObs(1,i));
125  }
126  }
127  return projections;
128  }
