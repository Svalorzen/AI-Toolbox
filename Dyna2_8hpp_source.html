219  template <typename M>
220  void Dyna2<M>::stepUpdateQ(const size_t s, const size_t a, const size_t s1, const size_t a1, const double rew) {
221  // We copy the traces from the permanent SARSAL to the transient one so
222  // that they will update their respective QFunctions in (nearly) the
223  // same way.
224  //
225  // Note that this is not quite the same as it is stated in the paper.
226  // Normally one would update only permanentLearning_, and transfer the
227  // exact same changes directly to the QFunction of transientLearning_.
228  //
229  // They differ since the QFunction inside each method are different,
230  // and so the updates won't exactly match. At the same time, after each
231  // reset (or end of episodes) the transient memory should reset to the
232  // permanent one, so this minor differences should go away.
233  //
234  // Ideally one would update directly the two QFunctions here, but this
235  // would basically require re-implementing SARSAL both here and in the
236  // batchUpdateQ method, which we avoid here for practicality.
237  transientLearning_.setTraces(permanentLearning_.getTraces());
238  permanentLearning_.stepUpdateQ(s, a, s1, a1, rew);
239  transientLearning_.stepUpdateQ(s, a, s1, a1, rew);
240  }

242  template <typename M>
243  void Dyna2<M>::batchUpdateQ(const size_t initS) {
244  // This clearing may not be needed if this is called after stepUpdateQ
245  // with the same s1 (since the set traces there will be correct then).
246  // We do it anyway in case this method is called in different settings
247  // and/or multiple times in a row.
248  transientLearning_.clearTraces();
249 
250  size_t s = initS;
251  size_t a = internalPolicy_->sampleAction(s);
252  for ( unsigned i = 0; i < N; ++i ) {
253  const auto [s1, rew] = model_.sampleSR(s, a);
254  const size_t a1 = internalPolicy_->sampleAction(s1);
255 
256  transientLearning_.stepUpdateQ(s, a, s1, a1, rew);
257 
258  if (model_.isTerminal(s1)) {
259  s = initS;
260  a = internalPolicy_->sampleAction(s);
261  } else {
262  s = s1;
263  a = a1;
264  }
265  }
266  }
