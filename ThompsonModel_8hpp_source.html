260  template <typename E>
261  void ThompsonModel<E>::sync(const size_t s, const size_t a) {
262  if constexpr (is_experience_eigen_v<E>) {
263  sampleDirichletDistribution(
264  // Here we add the Jeffreys prior
265  //
266  // Ideally this shouldn't allocate, as the casting and sum
267  // should simply create a wrapper Eigen object which is passed
268  // by reference, so should be still as efficient as doing it by
269  // hand.
270  experience_.getVisitsTable(a).row(s).array().template cast<double>() + 0.5,
271  rand_, transitions_[a].row(s)
272  );
273  } else {
274  // Sample manually
275  double sum = 0.0;
276  for (size_t s1 = 0; s1 < S; ++s1) {
277  // Here we add the Jeffreys prior
278  std::gamma_distribution<double> dist(experience_.getVisits(s, a, s1) + 0.5, 1.0);
279  transitions_[a](s, s1) = dist(rand_);
280  sum += transitions_[a](s, s1);
281  }
282  transitions_[a].row(s) /= sum;
283  }
284 
285  const auto visits = experience_.getVisitsSum(s, a);
286  const auto MLEReward = experience_.getReward(s, a);
287  const auto M2 = experience_.getM2(s, a);
288  if (visits < 2) {
289  // If we don't have enough info for the STD, we revert to MLE.
290  rewards_(s, a) = MLEReward;
291  } else {
292  std::student_t_distribution<double> dist(visits - 1);
293  rewards_(s, a) = MLEReward + dist(rand_) * std::sqrt(M2 / (visits * (visits - 1)));
294  }
295  }
