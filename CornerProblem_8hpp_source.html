8  // The gist of this problem is a small grid where
9  // the upper-left corner and the bottom-right corner
10  // are self-absorbing states. The agent can move in a
11  // top-left-down-right way, where each transition that
12  // is not self absorbing results in a reward penalty of -1.
13  // In addition the movements are not guaranteed: the
14  // agent succeeds only 80% of the time.
15  //
16  // Thus the agent needs to be able to find the shortest
17  // path to one of the self-absorbing states from every other
18  // state.
19  //

45  inline AIToolbox::MDP::Model makeCornerProblem(const GridWorld & grid, double stepUncertainty = 0.8) {
46  using namespace GridWorldEnums;
47 
48  size_t S = grid.getWidth() * grid.getHeight();
49  size_t A = 4;
50 
51  AIToolbox::DumbMatrix3D transitions(boost::extents[S][A][S]);
52  AIToolbox::DumbMatrix3D rewards(boost::extents[S][A][S]);
53 
54  for ( int x = 0; x < 4; ++x ) {
55  for ( int y = 0; y < 4; ++y ) {
56  auto s = grid(x,y);
57  if ( s == 0 || s == S-1 ) {
58  // Self absorbing states
59  for ( size_t a = 0; a < A; ++a )
60  transitions[s][a][s] = 1.0;
61  }
62  else {
63  for ( size_t a = 0; a < A; ++a ) {
64  auto s1 = grid.getAdjacent((Direction)a, s);
65  // If the move takes you outside the map, it doesn't do
66  // anything
67  if ( s == s1 ) transitions[s][a][s1] = 1.0;
68  else {
69  transitions[s][a][s1] = stepUncertainty;
70  transitions[s][a][s] = 1.0 - stepUncertainty;
71  }
72  rewards[s][a][s1] = -1.0;
73  }
74  }
75  }
76  }
77  return Model(S, A, transitions, rewards, 0.95);
78  }
79 }
