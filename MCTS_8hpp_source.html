192  template <typename M>
193  size_t MCTS<M>::sampleAction(const size_t a, const size_t s1, const unsigned horizon) {
194  auto & states = graph_.children[a].children;
195 
196  auto it = states.find(s1);
197  if ( it == states.end() )
198  return sampleAction(s1, horizon);
199 
200  // Here we need an additional step, because *it is contained by graph_.
201  // If we just move assign, graph_ is first going to delete everything it
202  // contains (included *it), and then we are going to move unallocated memory
203  // into graph_! So we move *it outside of the graph_ hierarchy, so that
204  // we can then assign safely.
205  { auto tmp = std::move(it->second); graph_ = std::move(tmp); }
206 
207  // We resize here in case we didn't have time to sample the new
208  // head node. In this case, the new head may not have children.
209  // This would break the UCT call.
210  graph_.children.resize(A);
211 
212  return runSimulation(s1, horizon);
213  }

228  template <typename M>
229  double MCTS<M>::simulate(StateNode & sn, const size_t s, const unsigned depth) {
230  // Head update
231  sn.N++;
232 
233  auto begin = std::begin(sn.children);
234  const size_t a = std::distance(begin, findBestBonusA(begin, std::end(sn.children), sn.N));
235 
236  auto [s1, rew] = model_.sampleSR(s, a);
237 
238  auto & aNode = sn.children[a];
239 
240  // We only go deeper if needed (maxDepth_ is always at least 1).
241  if ( depth + 1 < maxDepth_ && !model_.isTerminal(s1) ) {
242  const auto end = std::end(aNode.children);
243  auto it = aNode.children.find(s1);
244 
245  double futureRew;
246  if ( it == end ) {
247  // Touch node to create it
248  aNode.children[s1];
249  futureRew = rollout(s1, depth + 1);
250  }
251  else {
252  // Since most memory is allocated on the leaves,
253  // we do not allocate on node creation but only when
254  // we are actually descending into a node. If the node
255  // already has memory this should not do anything in
256  // any case.
257  it->second.children.resize(A);
258  futureRew = simulate( it->second, s1, depth + 1 );
259  }
260 
261  rew += model_.getDiscount() * futureRew;
262  }
263 
264  // Action update
265  aNode.N++;
266  aNode.V += ( rew - aNode.V ) / static_cast<double>(aNode.N);
267 
268  return rew;
269  }

272  double MCTS<M>::rollout(size_t s, unsigned depth) {
273  double rew = 0.0, totalRew = 0.0, gamma = 1.0;
274 
275  std::uniform_int_distribution<size_t> generator(0, A-1);
276  for ( ; depth < maxDepth_; ++depth ) {
277  std::tie( s, rew ) = model_.sampleSR( s, generator(rand_) );
278  totalRew += gamma * rew;
279 
280  if (model_.isTerminal(s))
281  return totalRew;
282 
283  gamma *= model_.getDiscount();
284  }
285  return totalRew;
286  }

295  template <typename Iterator>
296  Iterator MCTS<M>::findBestBonusA(Iterator begin, Iterator end, const unsigned count) {
297  // Count here can be as low as 1.
298  // Since log(1) = 0, and 0/0 = error, we add 1.0.
299  const double logCount = std::log(count + 1.0);
300  // We use this function to produce a score for each action. This can be easily
301  // substituted with something else to produce different POMCP variants.
302  const auto evaluationFunction = [this, logCount](const ActionNode & an){
303  return an.V + exploration_ * std::sqrt( logCount / an.N );
304  };
305 
306  auto bestIterator = begin++;
307  double bestValue = evaluationFunction(*bestIterator);
308 
309  for ( ; begin < end; ++begin ) {
310  double actionValue = evaluationFunction(*begin);
311  if ( actionValue > bestValue ) {
312  bestValue = actionValue;
313  bestIterator = begin;
314  }
315  }
316 
317  return bestIterator;
318  }
