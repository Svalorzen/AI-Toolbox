154 
155  Projecter project(model);
156  auto v = makeValueFunction(S); // TODO: May take user input
157 
158  unsigned timestep = 0;
159  const bool useTolerance = checkDifferentSmall(tolerance_, 0.0);
160  double variation = tolerance_ * 2; // Make it bigger
161  while ( timestep < horizon_ && ( !useTolerance || variation > tolerance_ ) ) {
162  ++timestep;
163 
164  auto projections = project(v[timestep-1]);
165 
166  // These are the good vectors, the ones that we are going to return for
167  // sure.
168  VList goodSupports;
169 
170  // We use this as a sorted linked list to handle all vectors, so it's
171  // easy to check whether we already have one, and also each vertex can
172  // keep references to them and we don't duplicate vectors all over the
173  // place.
174  SupportSet allSupports;
175 
176  std::vector<std::pair<Belief, double>> vertices;
177  std::unordered_set<Belief, boost::hash<Belief>> triedVertices;
178 
179  // For each corner belief, find its value and alphavector. Add the
180  // alphavectors in a separate list, remove duplicates. Note: In theory
181  // we must be able to find all alphas for each corner, not just a
182  // single best but all best. Cassandra does not do that though.. maybe
183  // we can avoid it? He uses the more powerful corner detection tho.
184  Belief corner(S); corner.fill(0.0);
185  for (size_t s = 0; s < S; ++s) {
186  corner[s] = 1.0;
187 
188  const auto [it, inserted] = allSupports.emplace(crossSumBestAtBelief(corner, projections));
189  if (inserted) goodSupports.push_back(*it);
190 
191  corner[s] = 0.0;
192  }
193 
194  // Now we find for all the alphavectors we have found, the vertices of
195  // the polytope that they created. These vertices will bootstrap the
196  // algorithm.
197  auto goodBegin = boost::make_transform_iterator(goodSupports.cbegin(), unwrap);
198  for (size_t i = 0; i < goodSupports.size(); ++i, ++goodBegin) {
199  // For each alpha, we find its vertices against the others.
200  IndexSkipMap map({i}, goodSupports);
201  const auto cbegin = boost::make_transform_iterator(map.cbegin(), unwrap);
202  const auto cend = boost::make_transform_iterator(map.cend(), unwrap);
203 
204  // Note that the range we pass here is made by a single vector.
205  auto newVertices = findVerticesNaive(goodBegin, goodBegin + 1, cbegin, cend);
206 
207  vertices.insert(std::end(vertices),
208  std::make_move_iterator(std::begin(newVertices)),
209  std::make_move_iterator(std::end(newVertices))
210  );
211  }
212 
213  do {
214  // For each corner, we find its true alphas and its best possible value.
215  // Then we compute the error between a corner's known true value and
216  // what we can do with the optimal alphas we already have.
217  // If the error is low enough, we don't need them. Otherwise we add
218  // them to the priority queue.
219  for (auto & vertex : vertices) {
220  if (triedVertices.find(vertex.first) != std::end(triedVertices))
221  continue;
222 
223  double trueValue;
224  auto support = crossSumBestAtBelief(vertex.first, projections, &trueValue);
225 
226  double currentValue = vertex.second;
227  // FIXME: As long as we use the naive way to find vertices,
228  // we can't really trust the values that come out as they
229  // may be lower than what we actually have. So we are
230  // forced to recompute their value.
231  const auto gsBegin = boost::make_transform_iterator(std::cbegin(goodSupports), unwrap);
232  const auto gsEnd = boost::make_transform_iterator(std::cend(goodSupports), unwrap);
233  findBestAtPoint(vertex.first, gsBegin, gsEnd, &currentValue);
234 
235  auto diff = trueValue - currentValue;
236  if (diff > tolerance_ && checkDifferentGeneral(diff, tolerance_)) {
237  auto it = allSupports.insert(std::move(support));
238  Vertex newVertex;
239  newVertex.belief = vertex.first;
240  newVertex.currentValue = currentValue;
241  newVertex.support = it.first;
242  newVertex.error = diff;
243  agenda_.push(std::move(newVertex));
244  }
245 
246  triedVertices.insert(std::move(vertex.first));
247  }
248 
249  if (agenda_.size() == 0)
250  break;
251 
252  Vertex best = agenda_.top();
253  agenda_.pop();
254 
255  AI_LOGGER(AI_SEVERITY_INFO, "Selected Vertex " << best.belief.transpose() << " as best, with support: " << best.support->values.transpose() << ", action: " << best.support->action);
256 
257  std::vector<Agenda::handle_type> verticesToRemove;
258 
259  // For each element in the agenda, we need to check whether any would
260  // be made useless by the new supports that best is bringing in. If so,
261  // we can remove them from the queue.
262  for (auto it = agenda_.begin(); it != agenda_.end(); ++it) {
263  // If so, *their* supports, plus the supports of best form the surface
264  // in which we need to find vertices.
265  if (it->belief.dot(best.support->values) > it->currentValue)
266  verticesToRemove.push_back(Agenda::s_handle_from_iterator(it));
267  }
268 
269  AI_LOGGER(AI_SEVERITY_DEBUG, "Removing " << verticesToRemove.size() << " vertices, as they are now obsolete.");
270  for (auto & h : verticesToRemove) {
271  agenda_.erase(h);
272  }
273 
274  // Find vertices between the best support of this belief and the list
275  // we already have.
276  const auto bestAddr = &(*best.support);
277  const auto supBegin = boost::make_transform_iterator(bestAddr, unwrap);
278  const auto supEnd = boost::make_transform_iterator(bestAddr + 1, unwrap);
279  const auto chkBegin = boost::make_transform_iterator(std::cbegin(goodSupports), unwrap);
280  const auto chkEnd = boost::make_transform_iterator(std::cend(goodSupports), unwrap);
281  vertices = findVerticesNaive(supBegin, supEnd, chkBegin, chkEnd);
282 
283  // We now can add the support for this vertex to the main list. We
284  // don't need checks here because we are guaranteed that we are
285  // improving the VList.
286  goodSupports.push_back(*best.support);
287  }
288  while (true);
289 
290  v.emplace_back(std::move(goodSupports));
291 
292  // Check convergence
293  if ( useTolerance ) {
294  variation = weakBoundDistance(v[timestep-1], v[timestep]);
295  }
296  }
297 
298  return std::make_tuple(useTolerance ? variation : 0.0, v);
299  }
