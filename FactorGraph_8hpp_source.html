281  {
282  // Try to take as much memory from the pool as possible.
283  auto oIt = std::begin(other.factorAdjacencies_);
284  while (factorAdjacencies_.size() < other.factorAdjacencies_.size()) {
285  if (factorAdjacenciesPool_.size() > 0) {

339  typename FactorGraph<FD>::FactorIt FactorGraph<FD>::getFactor(const Variables & variables) {
340  const auto found = findFactorByVariables(variableAdjacencies_[variables[0]].factors, variables);
341  if (found != variableAdjacencies_[variables[0]].factors.end())
342  return *found;
343 
344  FactorIt it;
345  if (!factorAdjacenciesPool_.size()) {
346  factorAdjacencies_.emplace_back(FactorNode());
347  it = --factorAdjacencies_.end();
348  } else {
349  factorAdjacencies_.splice(std::end(factorAdjacencies_), factorAdjacenciesPool_, std::begin(factorAdjacenciesPool_));
350  it = std::prev(factorAdjacencies_.end());
351  // We reset the data; just in case it's a vector we don't want to
352  // move but assign so that it does not clear already allocated
353  // memory.
354  auto tmp = FD{};
355  it->f_ = tmp;
356  }
357 
358  it->variables_ = variables;
359  for (const auto a : variables) {
360  auto & va = variableAdjacencies_[a];
361  va.factors.push_back(it);
362 
363  // Add *other* agents to vNeighbors
364  const auto mid = va.vNeighbors.size();
365  va.vNeighbors.reserve(mid + variables.size() - 1);
366 
367  for (size_t i = 0, j = 0; i < variables.size(); ) {
368  if (variables[i] == a) {
369  ++i;
370  } else if (j == mid || variables[i] < va.vNeighbors[j]) {
371  va.vNeighbors.push_back(variables[i]);
372  ++i;
373  } else {
374  if (variables[i] == va.vNeighbors[j])
375  ++i;
376  ++j;
377  }
378  }
379  std::inplace_merge(std::begin(va.vNeighbors), std::begin(va.vNeighbors)+mid, std::end(va.vNeighbors));
380  }

430  size_t FactorGraph<FD>::bestVariableToRemove(const Factors & F) const {
431  if (activeVariables_ == 0) return 0;
432 
433  // Find first active variable
434  size_t retval = 0;
435  while (!variableAdjacencies_[retval].active) ++retval;
436 
437  // Find the neighbors of this variable, and whether there's a factor
438  // with all of them.
439  const auto & vNeighbors = getVariables(retval);
440 
441  // We want the variable with the minimum size factor, where the factor
442  // already exists (so we don't have to allocate anything).
443  bool factorExists = false;
444  if (vNeighbors.size() > 0) {
445  const auto factorIt = findFactorByVariables(variableAdjacencies_[vNeighbors[0]].factors, vNeighbors);
446  factorExists = factorIt != std::end(variableAdjacencies_[vNeighbors[0]].factors);
447  }
448 
449  size_t minCost = F[retval];
450  for (auto n : vNeighbors)
451  minCost *= F[n];
452 
453  for (size_t next = retval + 1; next < variableAdjacencies_.size(); ++next) {
454  if (!variableAdjacencies_[next].active)
455  continue;
456 
457  const auto & vNeighbors = getVariables(next);
458 
459  bool newExists = false;
460  if (vNeighbors.size() > 0) {
461  const auto factorIt = findFactorByVariables(variableAdjacencies_[vNeighbors[0]].factors, vNeighbors);
462  newExists = factorIt != std::end(variableAdjacencies_[vNeighbors[0]].factors);
463  }
464 
465  // If we already have a factor, there's no point in looking at this
466  // variable.
467  if (!newExists && factorExists) continue;
468 
469  // Otherwise compute its cost
470  size_t newCost = F[next];
471  for (auto n : vNeighbors)
472  newCost *= F[n];
473 
474  // If we didn't have a factor, or the new cost is less than the old
475  // one, we select this variable.
476  if ((newExists && !factorExists) || (newCost < minCost)) {
477  retval = next;
478  minCost = newCost;
479  }
480  }
481  return retval;
482  }
