373  size_t FactorGraph<FD>::bestVariableToRemove(const Factors & F) const {
374  if (activeVariables_ == 0) return 0;
375 
376  // Find first active variable
377  size_t retval = 0;
378  while (!variableAdjacencies_[retval].active) ++retval;
379 
380  // Find the neighbors of this variable, and whether there's a factor
381  // with all of them.
382  const auto & vNeighbors = getVariables(retval);
383 
384  // We want the variable with the minimum size factor, where the factor
385  // already exists (so we don't have to allocate anything).
386  bool factorExists = false;
387  if (vNeighbors.size() > 0) {
388  const auto factorIt = findFactorByVariables(variableAdjacencies_[vNeighbors[0]].factors, vNeighbors);
389  factorExists = factorIt != std::end(variableAdjacencies_[vNeighbors[0]].factors);
390  }
391 
392  size_t minCost = F[retval];
393  for (auto n : vNeighbors)
394  minCost *= F[n];
395 
396  for (size_t next = retval + 1; next < variableAdjacencies_.size(); ++next) {
397  if (!variableAdjacencies_[next].active)
398  continue;
399 
400  const auto & vNeighbors = getVariables(next);
401 
402  bool newExists = false;
403  if (vNeighbors.size() > 0) {
404  const auto factorIt = findFactorByVariables(variableAdjacencies_[vNeighbors[0]].factors, vNeighbors);
405  newExists = factorIt != std::end(variableAdjacencies_[vNeighbors[0]].factors);
406  }
407 
408  // If we already have a factor, there's no point in looking at this
409  // variable.
410  if (!newExists && factorExists) continue;
411 
412  // Otherwise compute its cost
413  size_t newCost = F[next];
414  for (auto n : vNeighbors)
415  newCost *= F[n];
416 
417  // If we didn't have a factor, or the new cost is less than the old
418  // one, we select this variable.
419  if ((newExists && !factorExists) || (newCost < minCost)) {
420  retval = next;
421  minCost = newCost;
422  }
423  }
424  return retval;
425  }
