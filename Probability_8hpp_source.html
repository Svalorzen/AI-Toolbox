201  ProbabilityVector makeRandomProbability(const size_t S, G & generator) {
202  ProbabilityVector b(S);
203  double * bData = b.data();
204  // The way this works is that we're going to generate S-1 numbers in
205  // [0,1], and sort them with together with an implied 0.0 and 1.0, for
206  // a total of S+1 numbers.
207  //
208  // The output will be represented by the differences between each pair
209  // of numbers, after sorting the original vector.
210  //
211  // The idea is basically to take a unit vector and cut it up into
212  // random parts. The size of each part is the value of an entry of the
213  // output.
214 
215  // We must set the first element to zero even if we're later
216  // overwriting it. This is to avoid bugs in case the input S is one -
217  // in which case we should return a vector with a single element
218  // containing 1.0.
219  bData[0] = 0.0;
220  for ( size_t s = 0; s < S-1; ++s )
221  bData[s] = probabilityDistribution(generator);
222 
223  // Sort all but the implied last 1.0 which we'll add later.
224  std::sort(bData, bData + S - 1);
225 
226  // For each number, keep track of what was in the vector there, and
227  // transform it into the difference with its predecessor.
228  double helper1 = bData[0], helper2;
229  for ( size_t s = 1; s < S - 1; ++s ) {
230  helper2 = bData[s];
231  bData[s] -= helper1;
232  helper1 = helper2;
233  }
234  // The final one is computed with respect to the overall sum of 1.0.
235  bData[S-1] = 1.0 - helper1;
236 
237  return b;
238  }
