180  void PrioritizedSweeping<M>::stepUpdateQ(const size_t s, const size_t a) {
181  auto & values = vfun_.values;
182 
183  // Update q[s][a]
184  if constexpr(is_model_eigen_v<M>) {
185  qfun_(s,a) = model_.getRewardFunction().coeff(s, a) + model_.getTransitionFunction(a).row(s).dot(values * model_.getDiscount());
186  } else {
187  double newQValue = 0;
188  for ( size_t s1 = 0; s1 < S; ++s1 ) {
189  const double probability = model_.getTransitionProbability(s,a,s1);
190  if ( checkDifferentSmall( probability, 0.0 ) )
191  newQValue += probability * ( model_.getExpectedReward(s,a,s1) + model_.getDiscount() * values[s1] );
192  }
193  qfun_(s, a) = newQValue;
194  }
195 
196  double p = values[s];
197  {
198  // Update value and action
199  values[s] = qfun_.row(s).maxCoeff(&(vfun_.actions[s]));
200  }
201 
202  p = std::fabs(values[s] - p);
203 
204  for ( size_t ss = 0; ss < S; ++ss ) {
205  for ( size_t a = 0; a < A; ++a ) {
206  const double delta = p * model_.getTransitionProbability(ss,a,s);
207  // If it changed enough, we're going to update its parents.
208  if ( delta > theta_ ) {
209  const auto pair = std::make_pair(ss, a);
210  auto it = queueHandles_.find(pair);
211 
212  if (it != std::end(queueHandles_)) {
213  if ((*it->second).priority < delta) {
214  (*it->second).priority = delta;
215  queue_.increase(it->second);
216  }
217  } else {
218  queueHandles_[pair] = queue_.emplace(PriorityQueueElement{delta, pair});
219  }
220  }
221  }
222  }
223  }

226  void PrioritizedSweeping<M>::batchUpdateQ() {
227  for ( unsigned i = 0; i < N; ++i ) {
228  if ( queue_.empty() ) return;
229 
230  // The state we extract has been processed already

233  (void)p;
234 
235  queue_.pop();
236  queueHandles_.erase(pair);
237 
238  stepUpdateQ(pair.first, pair.second);
