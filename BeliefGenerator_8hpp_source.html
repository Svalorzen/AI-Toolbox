118  template <typename M>
119  void BeliefGenerator<M>::operator()(const size_t maxBeliefs, BeliefList * bl) const {
120  if ( !bl ) return;
121 
122  // Initialize all helper storage.
123  //
124  // We have:
125  //
126  // - bl: The belief list, which will contain beliefs we ever find,
127  // divided into two groups: the good ones (which we will return), and
128  // the bad ones. The good ones are further subdivided into the
129  // unproductive ones (which we do not want to sample from anymore as
130  // they are unlikely to produce anything new), and the productive
131  // ones.
132  // - seenObservations: This list contains, for each good belief, a list
133  // of action/observation pairs seen from it. This is used to avoid
134  // actually creating an updated belief if we have observed the a/o
135  // pair before.
136  // - unproductiveBeliefs: This list tracks the number of times we have
137  // tried to expand a particular belief. After a certain number amount
138  // of times we give up and signal that it is unproductive.
139  // - distances: This list contains, for each bad belief, its distance
140  // from the current good space. This is used to only pick the
141  // farthest beliefs when adding to the good set.
142  blp_ = bl;
143  auto & beliefs = *blp_;
144 
145  SeenObservations seenObservations;
146  seenObservations.resize(beliefs.size());
147 
148  std::vector<unsigned> unproductiveBeliefs;
149  unproductiveBeliefs.resize(beliefs.size());
150 
151  sop_ = &seenObservations;
152  up_ = &unproductiveBeliefs;
153 
154  beliefs.reserve(maxBeliefs);
155  seenObservations.reserve(maxBeliefs);
156  unproductiveBeliefs.reserve(maxBeliefs);
157 
158  std::vector<double> distances;
159  dp_ = &distances;
160 
161  // Since the original method of obtaining beliefs is stochastic,
162  // we keep trying for a while in case we don't find any new beliefs.
163  // However, for some problems (for example the Tiger problem) still
164  // this does not perform too well since the probability of finding
165  // a new belief (via action LISTEN) is pretty low.
166  size_t firstProductiveBelief = 0;
167  productiveBeliefs_ = goodBeliefsSize_ = allBeliefsSize_ = beliefs.size();
168 
169  unsigned randomBeliefsToAdd = 0;
170 
171  while ( goodBeliefsSize_ < maxBeliefs ) {
172  expandBeliefList(maxBeliefs, randomBeliefsToAdd, firstProductiveBelief);
173  if (goodBeliefsSize_ >= maxBeliefs) break;
174 
175  // Shift firstProductiveBelief to avoid checking the initial
176  // non-productive beliefs every single time.
177  for (size_t i = firstProductiveBelief; i < goodBeliefsSize_; ++i) {
178  if (unproductiveBeliefs[i] < retryLimit_) break;
179  else ++firstProductiveBelief;
180  }
181 
182  // Fill the missing if needed with random beliefs so we always have new stuff.
183  randomBeliefsToAdd = productiveBeliefs_ >= minProductiveBeliefs_ ? 0 : minProductiveBeliefs_ - productiveBeliefs_;
184  }
185  // Remove unused bad beliefs.
186  beliefs.resize(maxBeliefs);
187  }

189  template <typename M>
190  void BeliefGenerator<M>::expandBeliefList(const size_t max, const size_t randomBeliefsToAdd, const size_t firstProductiveBelief) const {
191  auto & bl = *blp_;
192  auto & seenObservations = *sop_;
193  auto & unproductiveBeliefs = *up_;
194  auto & distances = *dp_;
195 
196  // This is our optimistic estimate of how many beliefs we want to add
197  // this run; should be one per productive belief, or at least one per
198  // new random belief we are going to add.
199  //
200  // We refine this estimate later.
201  auto beliefsToAdd = std::max(randomBeliefsToAdd, productiveBeliefs_);
202 
203  // L1 distance
204  auto computeDistance = [](const Belief & lhs, const Belief & rhs) {
205  return (lhs - rhs).cwiseAbs().sum();
206  };
207 
208  // If we have some tentative test in the list, pop it off since we need
209  // to emplace the random beliefs.
210  if (allBeliefsSize_ < bl.size())
211  bl.pop_back();
212 
213  // Add the required random beliefs, computing distances for each.
214  for ( size_t i = 0; i < randomBeliefsToAdd; ++i) {
215  bl.emplace_back(makeRandomProbability(S, rand_));
216  ++allBeliefsSize_;
217 
218  // Compute distance for this belief.
219  distances.push_back(std::numeric_limits<double>::max());
220  for (size_t k = 0; k < goodBeliefsSize_; ++k) {
221  distances.back() = std::min(distances.back(), computeDistance(bl.back(), bl[k]));
222  }
223  }
224 
225  // We apply the discovery process to all beliefs we have approved as
226  // good. We start from the first productive one, since the others have
227  // already produced as much as they can.
228  for (size_t i = firstProductiveBelief; i < goodBeliefsSize_; ++i) {
229  // Skip this belief if it is unproductive.
230  auto & notFoundCounter = unproductiveBeliefs[i];
231  if (notFoundCounter >= retryLimit_) continue;
232 
233  auto & beliefObservations = seenObservations[i];
234  bool foundAnything = false;
235 
236  // Compute all new beliefs
237  for ( size_t a = 0; a < A; ++a ) {
238  updateBeliefPartial(model_, bl[i], a, &helper_);
239 
240  for (unsigned j = 0; j < triesPerRun_; ++j) {
241  // Sample state from belief, and generate an observation
242  // for it (given the current action)
243  const size_t s = sampleProbability(S, bl[i], rand_);
244 
245  size_t o;
246  std::tie(std::ignore, o, std::ignore) = model_.sampleSOR(s, a);
247 
248  // Check the new observation against the ones we have
249  // already produced for this belief. If we did, try again.
250  // Otherwise, mark it as seen.
251  if (beliefObservations.find({a,o}) != beliefObservations.end())
252  continue;
253 
254  beliefObservations.insert({a,o});
255  foundAnything = true;
256 
257  // Now we can update the belief. We use the last element of
258  // bl as temporary storage to avoid re-allocating when we
259  // don't need to.
260  if (allBeliefsSize_ == bl.size())
261  bl.emplace_back(S);
262 
263  updateBeliefPartialNormalized(model_, helper_, a, o, &bl.back());
264 
265  // Now check that the belief did not already exist in our
266  // list. If it did, we don't have to do anything else;
267  bool found = false;
268  for (size_t k = 0; k < allBeliefsSize_; ++k) {
269  if (checkEqualProbability(bl[k], bl.back())) {
270  found = true;
271  break;
272  }
273  }
274  if (found) continue;
275 
276  // Otherwise, the new belief is truly new. We keep it in
277  // the list and compute its distance. Note that we give an
278  // observation list only to the beliefs in the good set
279  // though (since we only sample those), so not yet to this
280  // one.
281  ++allBeliefsSize_;
282 
283  distances.push_back(std::numeric_limits<double>::max());
284  for (size_t k = 0; k < goodBeliefsSize_; ++k) {
285  distances.back() = std::min(distances.back(), computeDistance(bl.back(), bl[k]));
286  }
287  }
288  }
289  // We update the production counter for this belief, so we can skip
290  // the ones which are not needed.
291  if (!foundAnything) {
292  ++notFoundCounter;
293  // Mark it as unproductive if that's the case.
294  if (notFoundCounter == retryLimit_)
295  --productiveBeliefs_;
296  } else {
297  notFoundCounter = 0;
298  }
299  }
300  // Our optimistic estimane gets now scaled back by how many bad beliefs
301  // we actually have to make into good.
302  beliefsToAdd = std::min(beliefsToAdd, allBeliefsSize_ - goodBeliefsSize_);
303 
304  for (size_t i = 0; i < beliefsToAdd; ++i) {
305  assert((allBeliefsSize_ - goodBeliefsSize_) == distances.size());
306 
307  // Find furthest away. It's guaranteed to be new, so we add it to
308  // the good guys.
309  //
310  // We do a double swap here to avoid having to remove elements in
311  // the middle of distances; just saving a slight amount of work.
312  auto dBegin = std::begin(distances), dEnd = std::end(distances);
313  size_t id = std::distance( dBegin, std::max_element(dBegin, dEnd) );
314 
315  // Move the good one at the end
316  std::swap(distances[id], distances.back());
317  std::swap(bl[goodBeliefsSize_ + id], bl[allBeliefsSize_ - 1]);
318 
319  // We finally move the new selected belief in the right spot.
320  // Let's wait to do all the rest of the work until we know we have
321  // to.
322  std::swap(bl[goodBeliefsSize_], bl[allBeliefsSize_ - 1]);
323 
324  // Check if we are done.
325  ++goodBeliefsSize_;
326  if (goodBeliefsSize_ >= max) break;
327 
328  // If we are not done we:
329  //
330  // 1 - Remove the last entry from distances, which belonged to the
331  // now good belief.
332  // 2 - Add a seenObservations entry for the belief, since we can
333  // now sample from it.
334  // 3 - Recompute all remaining distances against the new good
335  // belief, as the space has changed.
336  distances.pop_back();
337  seenObservations.emplace_back();
338  unproductiveBeliefs.emplace_back();
339  ++productiveBeliefs_;
340 
341  for (size_t k = 0; k < distances.size(); ++k) {
342  distances[k] = std::min(distances[k], computeDistance(bl[goodBeliefsSize_ - 1], bl[goodBeliefsSize_ + k]));
343  }
344  }
345  }
346 }
